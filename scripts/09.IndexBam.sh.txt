#!/bin/sh
# ==============================================================================
# Script:       09.IndexBam.sh
# Description:  Index final deduplicated BAM files using SAMtools.
#
# Input:        Deduplicated 6x BAM files from 06.MarkDup.sh / 08.Downsample.sh
# Output:       BAI index files alongside each BAM in TARGET_DIR
# ==============================================================================

# ------------------------------------------------------------------------------
# SGE array job configuration
# ------------------------------------------------------------------------------
#$ -N IndexBam
#$ -cwd
#$ -l h_rt=2:00:00
#$ -l h_rss=4G
#$ -t 1-168
#$ -o o_files
#$ -e e_files

# ------------------------------------------------------------------------------
# Load required modules
# ------------------------------------------------------------------------------

. /etc/profile.d/modules.sh

module load roslin/samtools/1.9

# ------------------------------------------------------------------------------
# Path configuration
# ------------------------------------------------------------------------------

TARGET_DIR=/simone/pmax2023/out/06rmdup_bams/06x

# ------------------------------------------------------------------------------
# Parse sample for this array task
# ------------------------------------------------------------------------------

BAM=$(ls -1 "${TARGET_DIR}"/*.bam)
THIS_BAM=$(echo "${BAM}" | sed -n "${SGE_TASK_ID}p")

echo "=================================================="
echo "Task ID:    ${SGE_TASK_ID}"
echo "Input BAM:  ${THIS_BAM}"
echo "Host:       ${HOSTNAME}"
echo "Start time: $(date)"
echo "=================================================="

# ------------------------------------------------------------------------------
# Index BAM
# ------------------------------------------------------------------------------

samtools index "${THIS_BAM}"

echo "=================================================="
echo "Finished:   $(basename "${THIS_BAM}")"
echo "End time:   $(date)"
echo "=================================================="
